# Problem 9: Word Frequency Counter
# =====================================================================

def word_frequency(sentence):
    cleaned = sentence.translate(str.maketrans("", "", string.punctuation))

    # Lowercase everything so "Python" and "python" are treated the same,
    # then split into individual words on whitespace
    words = cleaned.lower().split()

    freq = {}
    for word in words:
        freq[word] = freq.get(word, 0) + 1

    # Display words in alphabetical order (sorted() on dict keys)
    for word in sorted(freq):
        print(f"{word} :{freq[word]}")

    # Word with the highest count
    most_repeated = max(freq, key=freq.get) if freq else None
    print("Most Repeated:")
    print(most_repeated)

    return freq  


def word_frequency():
    sentence = input("Enter a sentence: ")
    word_frequency(sentence)
